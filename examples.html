Node.js — discord.js + express

A full working bot that posts stats every 30 minutes and handles incoming vote webhooks. Uses discord.js v14 and express.

index.js
// npm install discord.js express node-fetch const { Client, GatewayIntentBits, EmbedBuilder } = require('discord.js'); const express = require('express'); const fetch = require('node-fetch'); // ──────────────── CONFIG ──────────────── const DISCORD_TOKEN = "YOUR_BOT_TOKEN"; const BOTLIY_API_TOKEN = "YOUR_BOTLIY_API_TOKEN"; const BOT_ID = "YOUR_BOT_ID"; const WEBHOOK_SECRET = "your_webhook_secret_here"; const VOTE_CHANNEL_ID = "123456789012345678"; // ──────────────────────────────────────── const client = new Client({ intents: [GatewayIntentBits.Guilds] }); const app = express(); app.use(express.json()); // ── Webhook handler ── app.post('/botliy/webhook', async (req, res) => { if (req.headers.authorization !== WEBHOOK_SECRET) return res.status(401).send("Unauthorized"); const { user } = req.body; const channel = client.channels.cache.get(VOTE_CHANNEL_ID); if (channel) { const embed = new EmbedBuilder() .setTitle("🎉 New Vote!") .setDescription(`<@${user}> just voted! Thanks ❤️`) .setColor("Blurple") .addFields( { name: "👤 Voter", value: `<@${user}>`, inline: true }, { name: "⏰ Time", value: `<t:${Math.floor(Date.now()/1000)}:R>`, inline: true } ); await channel.send({ embeds: [embed] }); } res.status(200).send("OK"); }); // ── Post stats every 30 min ── client.once('ready', () => { console.log(`Logged in as ${client.user.tag}`); app.listen(8999, () => console.log('Webhook server on :8999')); setInterval(async () => { try { await fetch(`https://botliy.online/api/bots/${BOT_ID}/stats`, { method: 'POST', headers: { Authorization: BOTLIY_API_TOKEN, 'Content-Type': 'application/json' }, body: JSON.stringify({ server_count: client.guilds.cache.size, shard_count: 1 }) }); } catch (e) { console.error("Stats error:", e); } }, 1800000); // 30 min }); client.login(DISCORD_TOKEN);
TypeScript

TypeScript — discord.js + express

The same integration with full TypeScript types. Uses an interface for the webhook payload and typed request/response objects.

index.ts
import { Client, GatewayIntentBits, EmbedBuilder, TextChannel } from 'discord.js'; import express, { Request, Response } from 'express'; import fetch from 'node-fetch'; const config = { DISCORD_TOKEN: "YOUR_BOT_TOKEN", BOTLIY_API_TOKEN: "YOUR_BOTLIY_API_TOKEN", BOT_ID: "YOUR_BOT_ID", WEBHOOK_SECRET: "your_webhook_secret_here", VOTE_CHANNEL_ID: "123456789012345678", }; interface WebhookPayload { bot: string; user: string; type: string; } const client = new Client({ intents: [GatewayIntentBits.Guilds] }); const app = express(); app.use(express.json()); app.post('/botliy/webhook', async (req: Request, res: Response) => { if (req.headers.authorization !== config.WEBHOOK_SECRET) return res.status(401).json({ error: 'Unauthorized' }); const { user } = req.body as WebhookPayload; const channel = client.channels.cache.get(config.VOTE_CHANNEL_ID) as TextChannel; if (channel) { await channel.send({ embeds: [ new EmbedBuilder() .setTitle("🎉 New Vote!") .setDescription(`<@${user}> just voted!`) .setColor("Blurple") ] }); } res.status(200).json({ status: 'ok' }); }); client.once('ready', () => { app.listen(8999); setInterval(async () => { await fetch(`https://botliy.online/api/bots/${config.BOT_ID}/stats`, { method: 'POST', headers: { Authorization: config.BOTLIY_API_TOKEN, 'Content-Type': 'application/json' }, body: JSON.stringify({ server_count: client.guilds.cache.size, shard_count: 1 }) }); }, 1800000); }); client.login(config.DISCORD_TOKEN);

Python — discord.py + Flask

Full implementation using discord.py for the bot and Flask for the webhook server, running concurrently via threads.

main.py
# pip install discord.py flask requests import discord, asyncio, requests from flask import Flask, request, jsonify from threading import Thread # ──────────────── CONFIG ──────────────── DISCORD_TOKEN = "YOUR_BOT_TOKEN" BOTLIY_API_TOKEN = "YOUR_BOTLIY_API_TOKEN" BOT_ID = "YOUR_BOT_ID" WEBHOOK_SECRET = "your_webhook_secret_here" VOTE_CHANNEL_ID = 123456789012345678 # ──────────────────────────────────────── intents = discord.Intents.default() client = discord.Client(intents=intents) app = Flask(__name__) @app.route('/botliy/webhook', methods=['POST']) def webhook(): if request.headers.get('Authorization') != WEBHOOK_SECRET: return jsonify({"error": "Unauthorized"}), 401 user_id = request.json.get("user") asyncio.run_coroutine_threadsafe( send_vote_msg(user_id), client.loop ) return jsonify({"status": "ok"}), 200 async def send_vote_msg(user_id): channel = client.get_channel(VOTE_CHANNEL_ID) if channel: embed = discord.Embed( title="🎉 New Vote!", description=f"<@{user_id}> just voted!", color=discord.Color.blurple() ) await channel.send(embed=embed) async def post_stats_loop(): await client.wait_until_ready() while not client.is_closed(): requests.post( f"https://botliy.online/api/bots/{BOT_ID}/stats", json={"server_count": len(client.guilds), "shard_count": 1}, headers={"Authorization": BOTLIY_API_TOKEN} ) await asyncio.sleep(1800) # 30 min @client.event async def on_ready(): client.loop.create_task(post_stats_loop()) def run_flask(): app.run(host="0.0.0.0", port=8999) if __name__ == "__main__": Thread(target=run_flask).start() client.run(DISCORD_TOKEN)

Go — discordgo + net/http

High-performance implementation using the standard library for HTTP and discordgo for the Discord bot.

main.go
package main import ( "bytes" "encoding/json" "fmt" "net/http" "time" "github.com/bwmarrin/discordgo" ) const ( BotToken = "YOUR_BOT_TOKEN" BotliyAPIToken = "YOUR_BOTLIY_API_TOKEN" BotID = "YOUR_BOT_ID" WebhookSecret = "your_webhook_secret_here" ChannelID = "123456789012345678" ) type VotePayload struct { User string `json:"user"` Bot string `json:"bot"` Type string `json:"type"` } func main() { dg, _ := discordgo.New("Bot " + BotToken) dg.Open() defer dg.Close() // Post stats every 30 min go func() { for { body, _ := json.Marshal(map[string]interface{}{ "server_count": len(dg.State.Guilds), "shard_count": 1, }) req, _ := http.NewRequest("POST", fmt.Sprintf("https://botliy.online/api/bots/%s/stats", BotID), bytes.NewBuffer(body), ) req.Header.Set("Authorization", BotliyAPIToken) req.Header.Set("Content-Type", "application/json") http.DefaultClient.Do(req) time.Sleep(30 * time.Minute) } }() // Webhook handler http.HandleFunc("/botliy/webhook", func(w http.ResponseWriter, r *http.Request) { if r.Header.Get("Authorization") != WebhookSecret { http.Error(w, "Unauthorized", http.StatusUnauthorized) return } var p VotePayload json.NewDecoder(r.Body).Decode(&p) dg.ChannelMessageSendEmbed(ChannelID, &discordgo.MessageEmbed{ Title: "🎉 New Vote!", Description: fmt.Sprintf("<@%s> just voted!", p.User), Color: 0xF59E0B, }) w.WriteHeader(http.StatusOK) }) http.ListenAndServe(":8999", nil) }
